```java
public class Ex10 {
  public static void main(String[] args) {
    // Loop from 0 to 9, Step 1
    for(int i = 0; i < 10; i++) {
      System.out.println(i);
    }

    for(int i = 0; i < 10; i += 2) {
      System.out.println(i);
    }

    for(int i = 10; i > 0; i--) {
      System.out.println(i);
    }

    for(int i = 10; i > 0; i -= 2) {
      System.out.println(i);
    }

    for(int i = 1; i < 257; i *= 2) {
      System.out.println(i);
    }

    for(int i = 1024; i > 0; i /= 2) {
      System.out.println(i);
    }
  }
}
```
